public class ApexP0Exercises {
    
    public Integer nFibonacci(Integer n){
       
        
        //first numbers in the sequence
        List<Integer> fib = new List<Integer>{1,1};
            
            for(Integer i = 2; i <= n; i ++){
                //add the numbers in the list to get the next number in sequence
                Integer nextFibNum = fib[i-1] + fib[i-2];
                //add result to the list
                fib.add(nextFibNum);
            }
        System.debug('the '+ n + ' number is:' + fib[n]);
        return n;
        
    }
}
